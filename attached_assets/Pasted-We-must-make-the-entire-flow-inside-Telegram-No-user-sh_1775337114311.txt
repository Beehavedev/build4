from deposit_helper import DepositHelper

deposit_helper = DepositHelper()

async def deposit(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Usage: /deposit <amount>  e.g. /deposit 50")
        return
    try:
        amount = float(context.args[0])
        instructions = deposit_helper.generate_deposit_instructions(amount)
        await update.message.reply_text(instructions, parse_mode='Markdown')
    except:
        await update.message.reply_text("Please provide a valid amount: /deposit 50")

async def confirm_deposit(update: Update, context: ContextTypes.DEFAULT_TYPE):
    # User can paste tx hash later for verification
    await update.message.reply_text("Send me the transaction hash after depositing and I'll verify it.\nOr just wait 30-60s and type /status")
