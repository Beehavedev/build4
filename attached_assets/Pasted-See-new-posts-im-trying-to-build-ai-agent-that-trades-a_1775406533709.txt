import pandas as pd
import ta  # technical analysis library

def simple_ai_strategy(df: pd.DataFrame):
    # EMA + RSI example (you can replace with LLM call)
    df['ema8'] = ta.trend.EMAIndicator(df['close'], window=8).ema_indicator()
    df['ema21'] = ta.trend.EMAIndicator(df['close'], window=21).ema_indicator()
    df['rsi'] = ta.momentum.RSIIndicator(df['close']).rsi()

    if df['ema8'].iloc[-1] > df['ema21'].iloc[-1] and df['rsi'].iloc[-1] < 70:
        return "BUY"
    elif df['ema8'].iloc[-1] < df['ema21'].iloc[-1] and df['rsi'].iloc[-1] > 30:
        return "SELL"
    return "HOLD"

import os
from aster_dex.futures import Frenchie as AsterClient  # official SDK (Frenchie alias)

class AsterTrader:
    def __init__(self):
        self.client = AsterClient(
            key=os.getenv("ASTER_API_KEY"),
            secret=os.getenv("ASTER_API_SECRET"),
            recvWindow=10000,      # helps with timestamp issues
        )

    def get_balance(self):
        """Returns wallet + futures balance (USDT, etc.)"""
        return self.client.balance()

    def get_positions(self):
        """Current open positions with unrealized PNL"""
        return self.client.positionRisk()  # or self.client.position() if available

    def get_account_info(self):
        """Full account snapshot (balances + unrealized PNL)"""
        return self.client.accountWithJoinMargin()  # best for PNL overview

    def get_recent_trades(self, symbol="BTCUSDT", limit=10):
        """Last filled trades"""
        return self.client.userTrades(symbol=symbol, limit=limit)

    def get_income_history(self, symbol=None, limit=10):
        """Realized PNL, funding fees, commissions (full PNL history)"""
        params = {"limit": limit}
        if symbol:
            params["symbol"] = symbol
        return self.client.income(**params)

    def get_market_data(self, symbol="BTCUSDT"):
        """Latest ticker for quick price"""
        # SDK supports public endpoints too
        return self.client.ticker(symbol=symbol)  # or use klines below

    def get_klines(self, symbol="BTCUSDT", interval="1m", limit=100):
        """OHLCV data for TA/strategy"""
        return self.client.klines(symbol=symbol, interval=interval, limit=limit)

    def place_order(self, symbol, side, quantity, leverage=10, order_type="MARKET", price=None):
        """Place market or limit order"""
        params = {
            "symbol": symbol,
            "side": side.upper(),          # BUY or SELL
            "type": order_type.upper(),
            "quantity": quantity,
            "leverage": leverage,
        }
        if order_type.upper() == "LIMIT" and price:
            params["price"] = price
            params["timeInForce"] = "GTC"
        return self.client.new_order(**params)

from urllib.parse import urlencode

class AsterTrader:
    def __init__(self):
        self.api_key = os.getenv("ASTER_API_KEY")
        self.secret = os.getenv("ASTER_API_SECRET")
        self.base_url = "https://api.asterdex.com"   # confirm exact base if different (often fapi.asterdex.com or similar)
        self.recv_window = 30000

    def _sign(self, params):
        if not self.secret:
            return ""
        query_string = urlencode(sorted(params.items()))
        signature = hmac.new(self.secret.encode('utf-8'), query_string.encode('utf-8'), hashlib.sha256).hexdigest()
        return signature

    def _request(self, method, endpoint, params=None, signed=False):
        url = self.base_url + endpoint
        headers = {"X-Api-Key": self.api_key} if self.api_key else {}
        timestamp = int(time.time() * 1000)
        
        if params is None:
            params = {}
        params['timestamp'] = timestamp
        params['recvWindow'] = self.recv_window

        if signed:
            params['signature'] = self._sign(params)

        try:
            if method.upper() == "GET":
                response = requests.get(url, params=params, headers=headers, timeout=10)
            else:
                response = requests.post(url, params=params, headers=headers, timeout=10)
            return response.json()
        except Exception as e:
            return {"error": str(e)}
