let cachedPairs: string[] = []
let pairsCachedAt = 0

export async function getActivePairs(): Promise<string[]> {
  // Cache for 1 hour — pairs don't change often
  if (cachedPairs.length && Date.now() - pairsCachedAt < 3_600_000) {
    return cachedPairs
  }
  
  try {
    const res = await http(BASE_PUBLIC).get('/fapi/v1/exchangeInfo')
    const symbols = res.data.symbols as any[]
    
    // Only TRADING status perpetual pairs with USDT
    cachedPairs = symbols
      .filter(s => 
        s.status === 'TRADING' && 
        s.quoteAsset === 'USDT' &&
        s.contractType === 'PERPETUAL'
      )
      .map(s => s.symbol)
    
    pairsCachedAt = Date.now()
    console.log(`[Aster] Loaded ${cachedPairs.length} active pairs`)
    return cachedPairs
  } catch (err) {
    console.error('[Aster] Failed to fetch pairs, using defaults')
    return [
      'BTCUSDT', 'ETHUSDT', 'BNBUSDT', 'SOLUSDT', 'ARBUSDT',
      'DOGEUSDT', 'XRPUSDT', 'ADAUSDT', 'AVAXUSDT', 'LINKUSDT',

      'NEARUSDT', 'APTUSDT', 'OPUSDT', 'INJUSDT', 'SUIUSDT'
    ]
  }
}

STEP 2 — Smart pair selection for agents

Don't scan all 337 pairs every tick — that's too expensive.
Instead, score pairs by opportunity and scan the best 10:

export async function getHighOpportunityPairs(
  limit = 10
): Promise<string[]> {
  try {
    // Get 24h ticker data for all pairs
    const res = await http(BASE_PUBLIC).get('/fapi/v1/ticker/24hr')
    const tickers = res.data as any[]
    
    // Score each pair by opportunity
    const scored = tickers
      .filter(t => t.symbol.endsWith('USDT'))
      .map(t => ({
        symbol: t.symbol,
        volume: parseFloat(t.quoteVolume),      // high volume = liquid
        change: Math.abs(parseFloat(t.priceChangePercent)), // moving = opportunity
        score: parseFloat(t.quoteVolume) * 
               Math.abs(parseFloat(t.priceChangePercent))
      }))
      .filter(t => t.volume > 1_000_000)        // min $1M daily volume
      .sort((a, b) => b.score - a.score)         // highest opportunity first
      .slice(0, limit)
      .map(t => t.symbol)
    
    console.log('[Aster] Top opportunity pairs:', scored)
    return scored
  } catch {
    return ['BTCUSDT', 'ETHUSDT', 'SOLUSDT', 'BNBUSDT', 'ARBUSDT',
            'DOGEUSDT', 'XRPUSDT', 'AVAXUSDT', 'LINKUSDT', 'MATICUSDT']
  }
}

// Before the tick loop, get today's best pairs
const pairsToScan = agent.tradingPair === 'ALL' 
  ? await getHighOpportunityPairs(10)  // scan top 10 movers
  : [agent.tradingPair]                // scan specific pair

// Scan each pair and pick the best setup
let bestSetup = null
let bestScore = 0

for (const pair of pairsToScan) {
  const setup = await analyzeSetup(pair, agent)
  if (setup.score > bestScore) {
    bestScore = setup.score
    bestSetup = { ...setup, pair }
  }
}

// Only trade the best opportunity found
if (bestSetup && bestSetup.score >= 7) {
  await executeTrade(bestSetup, agent)
}
