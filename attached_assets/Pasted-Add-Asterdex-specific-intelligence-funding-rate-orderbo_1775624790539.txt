def get_orderbook_imbalance(self, symbol: str, depth: int = 20) -> float:
    # Fetch orderbook and calculate buy/sell pressure imbalance (0 to 1)
    ob = self.exchange.fetch_order_book(symbol, limit=depth)
    bid_vol = sum([x[1] for x in ob['bids']])
    ask_vol = sum([x[1] for x in ob['asks']])
    return bid_vol / (bid_vol + ask_vol) if bid_vol + ask_vol > 0 else 0.5
