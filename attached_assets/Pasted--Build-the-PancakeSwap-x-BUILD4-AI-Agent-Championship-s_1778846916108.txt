
Build the PancakeSwap x BUILD4 AI Agent Championship system end to end. This is a live 7 day competition where BUILD4 users deploy agents that trade on PancakeSwap BSC. One strategy tweak allowed per agent per 24 hours. Top 5 agents by PnL at the end of day 7 win a share of $3,000 USDT in BNB. Everything is on chain, public, and verifiable.

export interface LeaderboardEntry {

  agentName: string

  currentBalance: number

  pnlPercent: number
  tradesCount: number
  tweaksUsedToday: number
  lastTweakAt: string | null
  strategyStyle: string
  isTop5: boolean

export async function updateLeaderboard(competitionId: string): Promise<void> {
  const entries = await db.competitionEntry.findMany({
    where: { competitionId, isDisqualified: false },
    include: { agent: true, user: true }
  });

  // Update balances from chain
  for (const entry of entries) {
    try {
      const balance = await getPancakeSwapBalance(entry.walletAddress);
      const pnlUsdt = balance - entry.startingBalance;
      const pnlPercent = (pnlUsdt / entry.startingBalance) * 100;

      await db.competitionEntry.update({
        where: { id: entry.id },
        data: { currentBalance: balance, pnlUsdt, pnlPercent }
      });
    } catch (err: any) {
      console.error(`[Leaderboard] Failed to update ${entry.agentName}:`, err.message);
    }
  }

  // Rank all entries
  const ranked = await db.competitionEntry.findMany({
    where: { competitionId, isDisqualified: false },
    orderBy: { pnlPercent: 'desc' }
  });

  for (let i = 0; i < ranked.length; i++) {
    await db.competitionEntry.update({
      where: { id: ranked[i].id },
      data: { rank: i + 1 }
    });

    // Snapshot every hour
    await db.competitionSnapshot.create({
      data: {
        entryId: ranked[i].id,
        balance: ranked[i].currentBalance,
        pnlUsdt: ranked[i].pnlUsdt,
        pnlPercent: ranked[i].pnlPercent,
        rank: i + 1,
        recordedAt: new Date()
      }
    });
  }

  console.log(`[Leaderboard] Updated ${ranked.length} entries`);
}

export async function getLeaderboard(competitionId: string): Promise<LeaderboardEntry[]> {
  const entries = await db.competitionEntry.findMany({
    where: { competitionId, isDisqualified: false },
    orderBy: { pnlPercent: 'desc' },
    take: 50,
    include: { user: true }
  });

  return entries.map((e, i) => ({
    rank: i + 1,
    agentName: e.agentName,
    ownerUsername: e.user?.username ?? 'Anonymous',
    walletAddress: e.walletAddress,
    startingBalance: e.startingBalance,
    currentBalance: e.currentBalance,
    pnlUsdt: e.pnlUsdt,
    pnlPercent: e.pnlPercent,
    tradesCount: e.tradesCount,
    tweaksUsedToday: e.tweaksUsedToday,
    lastTweakAt: e.lastTweakAt?.toISOString() ?? null,
    strategyStyle: deriveStrategyStyle(e.strategy as any),
    isTop5: i < 5
  }));
}

function deriveStrategyStyle(strategy: any): string {
  if (!strategy) return 'Balanced';
  const leverage = strategy.maxLeverage ?? 1;
  const risk = strategy.riskTolerance ?? 'medium';
  if (leverage >= 10) return 'Aggressive';
  if (leverage >= 5) return 'Momentum';
  if (risk === 'low') return 'Conservative';
  return 'Balanced';
}

export interface StrategyTweak {
  maxLeverage: number          // 1 to 20
  riskTolerance: 'low' | 'medium' | 'high'
  preferredPairs: string[]     // e.g. ['BTCUSDT', 'BNBUSDT']
  signalWeights: {
    momentum: number           // 0 to 100
    meanReversion: number      // 0 to 100
    volumeBreakout: number     // 0 to 100
  }
  maxPositionSizePercent: number // 1 to 50
  stopLossPercent: number        // 1 to 20
  takeProfitPercent: number      // 1 to 100
}

export async function applyTweak(
  entryId: string,
  newStrategy: StrategyTweak,
  reasoning: string
): Promise<{ success: boolean; message: string }> {
  const entry = await db.competitionEntry.findUnique({
    where: { id: entryId }
  });

  if (!entry) return { success: false, message: 'Entry not found' };
  if (entry.isDisqualified) return { success: false, message: 'Agent is disqualified' };

  // Check tweak window — resets at midnight UTC
  if (entry.tweaksUsedToday >= 1) {
    const nextReset = getNextMidnightUTC();
    const hoursUntilReset = Math.ceil((nextReset.getTime() - Date.now()) / (1000 * 60 * 60));
    return {
      success: false,
      message: `Tweak already used today. Next window opens in ${hoursUntilReset} hours at midnight UTC.`
    };
  }

  // Validate strategy values
  const validation = validateStrategy(newStrategy);
  if (!validation.valid) {
    return { success: false, message: validation.reason ?? 'Invalid strategy' };
  }

  // Save tweak
  await db.competitionTweak.create({
    data: {
      entryId,
      previousStrategy: entry.strategy as any,
      newStrategy: newStrategy as any,
      reasoning,
      appliedAt: new Date()
    }
  });

  await db.competitionEntry.update({
    where: { id: entryId },
    data: {
      strategy: newStrategy as any,
      tweaksUsedToday: { increment: 1 },
      lastTweakAt: new Date()
    }
  });

  console.log(`[TweakManager] ${entry.agentName} tweaked strategy: ${reasoning}`);
  return { success: true, message: 'Strategy updated. Agent will apply new settings on next tick.' };
}

function validateStrategy(strategy: StrategyTweak): { valid: boolean; reason?: string } {
  if (strategy.maxLeverage < 1 || strategy.maxLeverage > 20)
    return { valid: false, reason: 'Max leverage must be between 1 and 20' };
  if (strategy.maxPositionSizePercent < 1 || strategy.maxPositionSizePercent > 50)
    return { valid: false, reason: 'Max position size must be between 1% and 50%' };
  if (strategy.stopLossPercent < 1 || strategy.stopLossPercent > 20)
    return { valid: false, reason: 'Stop loss must be between 1% and 20%' };
  if (strategy.preferredPairs.length === 0)
    return { valid: false, reason: 'Must select at least one trading pair' };
  return { valid: true };
}

function getNextMidnightUTC(): Date {
  const now = new Date();
  const midnight = new Date();
  midnight.setUTCHours(24, 0, 0, 0);
  return midnight;
}

STEP 4 — PANCAKESWAP INTEGRATION
File: src/services/competition/pancakeSwap.ts
typescriptimport { ethers } from 'ethers';

const PANCAKE_ROUTER = '0x10ED43C718714eb63d5aA57B78B54704E256024E';
const WBNB = '0xbb4CdB9CBd36B01bD1cBaEBF2De08d9173bc095c';
const USDT_BSC = '0x55d398326f99059fF775485246999027B3197955';

const ROUTER_ABI = [
  'function swapExactTokensForTokens(uint amountIn, uint amountOutMin, address[] calldata path, address to, uint deadline) external returns (uint[] memory amounts)',
  'function getAmountsOut(uint amountIn, address[] calldata path) external view returns (uint[] memory amounts)'
];

const ERC20_ABI = [
  'function approve(address spender, uint256 amount) external returns (bool)',
  'function balanceOf(address account) external view returns (uint256)',
  'function allowance(address owner, address spender) external view returns (uint256)'
];

export async function getPancakeSwapBalance(walletAddress: string): Promise<number> {
  const provider = new ethers.JsonRpcProvider(process.env.BSC_RPC_URL);
  const usdt = new ethers.Contract(USDT_BSC, ERC20_ABI, provider);
  const balance = await usdt.balanceOf(walletAddress);
  return parseFloat(ethers.formatUnits(balance, 18));
}

export async function executeSwap(
  privateKey: string,
  tokenIn: string,
  tokenOut: string,
  amountInUsdt: string,
  slippagePct: number = 2
): Promise<{ txHash: string; amountOut: string }> {
  const provider = new ethers.JsonRpcProvider(process.env.BSC_RPC_URL);
  const wallet = new ethers.Wallet(privateKey, provider);
  const router = new ethers.Contract(PANCAKE_ROUTER, ROUTER_ABI, wallet);
  const tokenInContract = new ethers.Contract(tokenIn, ERC20_ABI, wallet);

  const amountIn = ethers.parseUnits(amountInUsdt, 18);
  const path = [tokenIn, WBNB, tokenOut].filter((v, i, a) => a.indexOf(v) === i);

  // Get expected output
  const amounts = await router.getAmountsOut(amountIn, path);
  const expectedOut = amounts[amounts.length - 1];
  const minOut = expectedOut * BigInt(100 - slippagePct) / 100n;

  // Approve if needed
  const allowance = await tokenInContract.allowance(wallet.address, PANCAKE_ROUTER);
  if (allowance < amountIn) {
    const approveTx = await tokenInContract.approve(PANCAKE_ROUTER, ethers.MaxUint256);
    await approveTx.wait();
  }

  // Execute swap
  const deadline = Math.floor(Date.now() / 1000) + 300;
  const tx = await router.swapExactTokensForTokens(
    amountIn, minOut, path, wallet.address, deadline
  );
  const receipt = await tx.wait();

  return {
    txHash: receipt.hash,
    amountOut: ethers.formatUnits(expectedOut, 18)
  };
}

export async function tickCompetitionAgent(entryId: string): Promise<void> {
  const entry = await db.competitionEntry.findUnique({
    where: { id: entryId },
    include: { user: true }
  });

  if (!entry || entry.isDisqualified) return;

  const strategy = entry.strategy as any;
  const pairs = strategy.preferredPairs ?? ['BTCUSDT', 'BNBUSDT'];

  // Build market context for this agent
  const marketData = await buildMarketContext(pairs);
  const balance = await getPancakeSwapBalance(entry.walletAddress);
  const pnlPercent = ((balance - entry.startingBalance) / entry.startingBalance) * 100;

  const response = await client.messages.create({
    model: 'claude-sonnet-4-20250514',
    max_tokens: 500,
    system: `You are an autonomous AI trading agent competing in the PancakeSwap x BUILD4 AI Agent Championship.
You have 7 days to maximize PnL percentage. All trades execute on PancakeSwap BSC.
Your strategy settings: ${JSON.stringify(strategy)}
Current balance: $${balance.toFixed(2)} USDT
Current PnL: ${pnlPercent.toFixed(2)}%
Respond only in JSON.`,
    messages: [{
      role: 'user',
      content: `Market data: ${marketData}

Decide your next action. Options:
HOLD — do nothing this tick
BUY — buy a token with USDT
SELL — sell a token back to USDT

Respond in JSON:
{
  "action": "HOLD" or "BUY" or "SELL",
  "pair": "e.g. BNBUSDT" or null,
  "amountPercent": number 1 to ${strategy.maxPositionSizePercent} or null,
  "reasoning": "one line max"
}`
    }]
  });

  const text = response.content[0].type === 'text' ? response.content[0].text : '{}';
  const decision = JSON.parse(text.replace(/```json|```/g, '').trim());

  console.log(`[CompAgent] ${entry.agentName} — ${decision.action} ${decision.pair ?? ''} | ${decision.reasoning}`);

  if (decision.action === 'HOLD') return;

  if (decision.action === 'BUY' && decision.pair && decision.amountPercent) {
    try {
      const amountUsdt = (balance * decision.amountPercent / 100).toFixed(2);
      const privateKey = await getPrivateKey(entry.userId);
      const result = await executeSwap(privateKey, USDT_BSC, getTokenAddress(decision.pair), amountUsdt);

      await db.competitionEntry.update({
        where: { id: entryId },
        data: { tradesCount: { increment: 1 } }
      });

      console.log(`[CompAgent] ${entry.agentName} BUY executed | tx: ${result.txHash}`);
    } catch (err: any) {
      console.error(`[CompAgent] ${entry.agentName} BUY failed:`, err.message);
    }
  }
}

async function buildMarketContext(pairs: string[]): Promise<string> {
  const data: string[] = [];
  for (const pair of pairs.slice(0, 3)) {
    try {
      const candles = await getOHLC(pair, 0, '1h', 24);
      const last = candles[candles.length - 1];
      const prev = candles[candles.length - 2];
      const change = ((last.close - prev.close) / prev.close * 100).toFixed(2);
      data.push(`${pair}: $${last.close} (${change}% 1h)`);
    } catch {
      data.push(`${pair}: data unavailable`);
    }
  }
  return data.join(' | ');
}

async function getPrivateKey(userId: string): Promise<string> {
  const wallet = await db.wallet.findFirst({
    where: { userId, chain: 'BSC', isActive: true }
  });
  if (!wallet?.privateKey) throw new Error('No wallet found');
  return wallet.privateKey;
}
