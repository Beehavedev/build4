Build a complete autonomous token launch intelligence system for BUILD4. This system runs continuously, scores narratives across five dimensions, and launches tokens on four.meme when the evidence is strong enough. It is not a meme launcher. It is an alpha detection engine that expresses its conclusions as tokens.

ARCHITECTURE
src/
  services/
    signalScorer/
      index.ts              — orchestrator, runs every 30 min
      noveltyScorer.ts      — checks DexScreener for existing tokens
      velocityScorer.ts     — measures acceleration across X, Telegram, Reddit, Google
      verifiabilityScorer.ts — validates signal against on-chain and official sources
      communityFitScorer.ts  — scores meme potential and BSC community alignment
      timingScorer.ts        — estimates window before narrative goes mainstream
    launchPacket.ts         — assembles full launch spec before any transaction
    fourMemeLauncher.ts     — executes the actual launch on four.meme
    launchTracker.ts        — tracks every launched token for 7 days post launch
    launchFeedback.ts       — retrospective scoring to improve future launches
  agents/
    tokenLaunchAgent.ts     — Claude Sonnet decision layer
  routes/
    launchProposals.ts      — admin endpoints to view queue and override
scripts/
  runSignalScorer.ts        — manual trigger for testing

export interface SignalScore {
  novelty: number        // 0 to 20
  velocity: number       // 0 to 20
  verifiability: number  // 0 to 20
  communityFit: number   // 0 to 20
  timing: number         // 0 to 20
  total: number          // 0 to 100
  breakdown: string      // human readable explanation
  vetoed: boolean        // true if verifiability under 6
  vetoReason?: string
}

export async function runSignalScorerLoop(): Promise<void> {
  console.log('[SignalScorer] Starting scan cycle');

  // Pull candidates from all sources in parallel
  const candidates = await gatherCandidates();
  console.log(`[SignalScorer] Evaluating ${candidates.length} candidates`);

  for (const candidate of candidates) {
    try {
      await evaluateAndAct(candidate);
      await sleep(2000); // rate limit between evaluations
    } catch (err: any) {
      console.error(`[SignalScorer] Failed on ${candidate.keyword}:`, err.message);
    }
  }

  console.log('[SignalScorer] Cycle complete');
}

async function evaluateAndAct(candidate: NarrativeCandidate): Promise<void> {
  // Check if already evaluated recently
  const recentEval = await db.launchProposal.findFirst({
    where: {
      keyword: candidate.keyword,
      evaluatedAt: { gte: new Date(Date.now() - 6 * 60 * 60 * 1000) }
    }
  });
  if (recentEval) return;

  // Score all five dimensions in parallel
  const [novelty, velocity, verifiability, communityFit, timing] = await Promise.all([
    scoreNovelty(candidate),
    scoreVelocity(candidate),
    scoreVerifiability(candidate),
    scoreCommunityFit(candidate),
    scoreTiming(candidate)
  ]);

  const total = novelty + velocity + verifiability + communityFit + timing;
  const vetoed = verifiability < 6;

  const score: SignalScore = {
    novelty, velocity, verifiability, communityFit, timing, total,
    breakdown: `Novelty ${novelty} | Velocity ${velocity} | Verifiability ${verifiability} | Community ${communityFit} | Timing ${timing}`,
    vetoed,
    vetoReason: vetoed ? `Verifiability ${verifiability}/20 is below minimum threshold of 6` : undefined
  };

  // Log to DB
  await db.launchProposal.create({
    data: {
      keyword: candidate.keyword,
      description: candidate.description,
      sourceType: candidate.sourceType,
      rawData: JSON.stringify(candidate.rawData),
      scores: JSON.stringify(score),
      totalScore: total,
      status: vetoed ? 'vetoed' : total >= 70 ? 'approved' : 'rejected',
      evaluatedAt: new Date()
    }
  });

  console.log(`[SignalScorer] ${candidate.keyword} — Score: ${total}/100 | ${score.breakdown} | ${vetoed ? 'VETOED' : total >= 70 ? 'APPROVED' : 'REJECTED'}`);

  if (vetoed) {
    await notifyAdmin(`[SignalScorer] VETOED: ${candidate.keyword}\nReason: ${score.vetoReason}`);
    return;
  }

  if (total < 70) return;

  // Pass to Claude for final decision and launch packet assembly
  await handleApprovedCandidate(candidate, score);
}

async function handleApprovedCandidate(
  candidate: NarrativeCandidate,
  score: SignalScore
): Promise<void> {
  const packet = await assembleLaunchPacket(candidate, score);

  if (!packet) {
    console.log(`[SignalScorer] ${candidate.keyword} — packet assembly failed, skipping`);
    return;
  }

  // Determine launch tier
  const tier = score.total >= 90 ? 'priority' :
               score.total >= 80 ? 'standard' : 'soft';

  const liquidityMap = { priority: '300', standard: '150', soft: '50' };
  packet.initialLiquidityUsdt = liquidityMap[tier];

  console.log(`[SignalScorer] ${candidate.keyword} — ${tier.toUpperCase()} LAUNCH | liquidity: $${packet.initialLiquidityUsdt}`);

  await executeLaunch(packet, tier);
}

export async function scoreNovelty(candidate: NarrativeCandidate): Promise<number> {
  try {
    const keyword = encodeURIComponent(candidate.keyword.toLowerCase());
    const { data } = await axios.get(
      `https://api.dexscreener.com/latest/dex/search?q=${keyword}`,
      { timeout: 5000 }
    );

    const bscPairs = (data.pairs ?? []).filter((p: any) => p.chainId === 'bsc');

    if (bscPairs.length === 0) {
      console.log(`[Novelty] ${candidate.keyword} — no existing token: 20/20`);
      return 20;
    }

    // Check if existing tokens have meaningful traction
    const strongTokens = bscPairs.filter((p: any) => {
      const age = Date.now() - new Date(p.pairCreatedAt ?? 0).getTime();
      const ageHours = age / (1000 * 60 * 60);
      return (p.volume?.h24 ?? 0) > 50000 && ageHours < 72;
    });

    if (strongTokens.length > 0) {
      console.log(`[Novelty] ${candidate.keyword} — strong token exists: 0/20`);
      return 0;
    }

    // Weak or old tokens exist — partial opportunity
    const weakTokens = bscPairs.filter((p: any) => (p.volume?.h24 ?? 0) < 50000);
    if (weakTokens.length > 0) {
      console.log(`[Novelty] ${candidate.keyword} — weak token exists: 8/20`);
      return 8;
    }

    return 12;
  } catch {
    console.log(`[Novelty] ${candidate.keyword} — DexScreener error, defaulting to 10`);
    return 10;
  }
}

async function scoreXVelocity(keyword: string): Promise<number> {
  if (!process.env.TWITTER_BEARER_TOKEN) return 8; // neutral fallback

  try {
    const headers = { Authorization: `Bearer ${process.env.TWITTER_BEARER_TOKEN}` };
    const query = encodeURIComponent(`${keyword} -is:retweet lang:en`);

    const [recent, older] = await Promise.all([
      axios.get(`https://api.twitter.com/2/tweets/search/recent?query=${query}&max_results=100&start_time=${hoursAgo(1)}`, { headers }),
      axios.get(`https://api.twitter.com/2/tweets/search/recent?query=${query}&max_results=100&start_time=${hoursAgo(6)}&end_time=${hoursAgo(1)}`, { headers })
    ]);

    const recentCount = recent.data.meta?.result_count ?? 0;
    const olderCount = older.data.meta?.result_count ?? 0;

    if (olderCount === 0) return recentCount > 10 ? 20 : 10;

    const acceleration = recentCount / (olderCount / 5); // normalize to same time window
    if (acceleration > 5) return 20;
    if (acceleration > 3) return 15;
    if (acceleration > 1.5) return 10;
    if (acceleration > 1) return 5;
    return 0;
  } catch {
    return 8;
  }
}

typescriptimport axios from 'axios';
import { NarrativeCandidate } from './index';

export async function scoreVerifiability(candidate: NarrativeCandidate): Promise<number> {
  let score = 0;
  const evidence: string[] = [];

  // On-chain verification via BSCScan
  const onChainScore = await checkOnChain(candidate);
  score += onChainScore;
  if (onChainScore > 0) evidence.push(`on-chain: ${onChainScore}pts`);

  // Official source verification
  const officialScore = await checkOfficialSources(candidate);
  score += officialScore;
  if (officialScore > 0) evidence.push(`official: ${officialScore}pts`);

  // Multi-source cross verification
  const crossScore = await crossVerify(candidate);
  score += crossScore;
  if (crossScore > 0) evidence.push(`cross-verified: ${crossScore}pts`);

  const total = Math.min(20, score);
  console.log(`[Verifiability] ${candidate.keyword} — ${total}/20 | ${evidence.join(', ')}`);
  return total;
}

async function checkOnChain(candidate: NarrativeCandidate): Promise<number> {
  if (!process.env.BSCSCAN_API_KEY) return 0;

  try {
    // For whale anomaly signals check the specific wallet activity
    if (candidate.sourceType === 'whale_anomaly' && candidate.rawData?.walletAddress) {
      const { data } = await axios.get(`https://api.bscscan.com/api`, {
        params: {
          module: 'account',
          action: 'tokentx',
          address: candidate.rawData.walletAddress,
          apikey: process.env.BSCSCAN_API_KEY,
          sort: 'desc',
          page: 1,
          offset: 10
        }
      });

      if (data.result?.length > 0) return 8;
    }

    // For volume anomalies verify the contract exists and has real activity
    if (candidate.sourceType === 'volume_anomaly' && candidate.rawData?.contractAddress) {
      const { data } = await axios.get(`https://api.bscscan.com/api`, {
        params: {
          module: 'contract',
          action: 'getabi',
          address: candidate.rawData.contractAddress,
          apikey: process.env.BSCSCAN_API_KEY
        }
      });

      if (data.status === '1') return 6;
    }

    return 0;
  } catch {
    return 0;
  }
}

async function checkOfficialSources(candidate: NarrativeCandidate): Promise<number> {
  // AI milestones — check if a known official source published this
  if (candidate.sourceType === 'ai_milestone') {
    const officialDomains = [
      'openai.com', 'anthropic.com', 'deepmind.com',
      'mistral.ai', 'meta.com', 'google.com'
    ];
    const sourceUrl = candidate.rawData?.sourceUrl ?? '';
    if (officialDomains.some(d => sourceUrl.includes(d))) return 8;
    if (sourceUrl.includes('github.com')) return 6;
    return 2;
  }

  // Prediction winner — resolution is on-chain and objective
  if (candidate.sourceType === 'prediction_winner') return 8;

  // Geopolitical — check Reuters or Bloomberg
  if (candidate.sourceType === 'geopolitical') {
    const sourceUrl = candidate.rawData?.sourceUrl ?? '';
    if (sourceUrl.includes('reuters.com') || sourceUrl.includes('bloomberg.com')) return 8;
    if (sourceUrl.includes('bbc.com') || sourceUrl.includes('ap.org')) return 6;
    return 2;
  }

  return 3;
}

async function crossVerify(candidate: NarrativeCandidate): Promise<number> {
  const sourceCount = candidate.rawData?.sourceCount ?? 1;
  if (sourceCount >= 4) return 4;
  if (sourceCount >= 3) return 3;
  if (sourceCount >= 2) return 2;
  return 0;
}

typescriptimport { NarrativeCandidate } from './index';

export async function scoreTiming(candidate: NarrativeCandidate): Promise<number> {
  let score = 20;
  const penalties: string[] = [];

  // Estimate hours until mainstream based on current velocity
  const hoursUntilMainstream = estimateMainstreamTime(candidate);

  if (hoursUntilMainstream > 24) score = 20;
  else if (hoursUntilMainstream > 12) score = 15;
  else if (hoursUntilMainstream > 6) score = 8;
  else if (hoursUntilMainstream > 2) score = 3;
  else { score = 0; penalties.push('already mainstream'); }

  // Weekend penalty
  const day = new Date().getDay();
  if (day === 0 || day === 6) {
    score = Math.max(0, score - 3);
    penalties.push('weekend minus 3');
  }

  // Market panic penalty — check BTC 4h change
  const btcChange = candidate.rawData?.btcChange4h ?? 0;
  if (btcChange < -5) {
    score = Math.max(0, score - 5);
    penalties.push(`BTC down ${btcChange.toFixed(1)}% minus 5`);
  }

  const total = Math.min(20, score);
  console.log(`[Timing] ${candidate.keyword} — ${total}/20 | ~${hoursUntilMainstream}h to mainstream | ${penalties.join(', ') || 'no penalties'}`);
  return total;
}

function estimateMainstreamTime(candidate: NarrativeCandidate): number {
  const velocity = candidate.rawData?.xPostsLastHour ?? 0;

  // Empirical thresholds based on historical meme velocity patterns
  if (velocity > 5000) return 1;   // already viral
  if (velocity > 1000) return 3;
  if (velocity > 500) return 8;
  if (velocity > 100) return 16;
  if (velocity > 50) return 30;
  return 72; // very early, maximum window
}

export async function executeLaunch(
  packet: LaunchPacket,
  tier: LaunchTier
): Promise<void> {
  console.log(`[Launcher] Executing ${tier} launch: ${packet.tokenName} ($${packet.ticker})`);

      narrativeHook: packet.narrativeHook,

      scoreTotal: packet.score.total,
      scoreBreakdown: packet.scoreBreakdown,
      initialLiquidityUsdt: parseFloat(packet.initialLiquidityUsdt),
      tier,
      status: 'pending',
      telegramCaption: packet.telegramCaption,
      xPost: packet.xPost,
      fortyTwoQuestion: packet.fortyTwoMarketQuestion,
      expiryLogic: packet.expiryLogic,
      launchedAt: new Date()
    }
  });

  try {
    // Call your existing four.meme launch service
    // This should already exist in your codebase from the /launch command
    const { launchToken } = await import('./fourMeme');

    const result = await launchToken({
      name: packet.tokenName,
      symbol: packet.ticker,
      description: packet.narrativeHook,
      initialLiquidityUsdt: packet.initialLiquidityUsdt
    });

    // Update DB with result
    await db.tokenLaunch.update({
      where: { id: launch.id },
      data: {
        status: 'live',
        contractAddress: result.contractAddress,
        txHash: result.txHash,
        pairAddress: result.pairAddress
      }
    });

    console.log(`[Launcher] LIVE: ${packet.tokenName} | contract: ${result.contractAddress} | tx: ${result.txHash}`);

    // Notify based on tier
    const message = buildLaunchMessage(packet, result, tier);

    if (tier === 'priority') {
      await notifyChannel(message); // broadcast to all users
      await notifyAdmin(message);
    } else if (tier === 'standard') {
      await notifyAdmin(message);
    } else {
      console.log(`[Launcher] Soft launch — monitoring before promoting`);
      // Check again in 2 hours, promote if holding
      setTimeout(() => evaluateSoftLaunch(launch.id, message), 2 * 60 * 60 * 1000);
    }

    // If score above 85 and 42.space question exists, create market proposal
    if (packet.score.total >= 85 && packet.fortyTwoMarketQuestion) {
      await propose42Market(packet, result.contractAddress);
    }

  } catch (err: any) {
    console.error(`[Launcher] Launch failed for ${packet.tokenName}:`, err.message);
    await db.tokenLaunch.update({
      where: { id: launch.id },
      data: { status: 'failed', failReason: err.message }
    });
    await notifyAdmin(`[Launcher] FAILED: ${packet.tokenName}\nReason: ${err.message}`);
  }
}

export async function runLaunchTrackerLoop(): Promise<void> {
  const activeLaunches = await db.tokenLaunch.findMany({
    where: {
      status: 'live',
      launchedAt: { gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000) }
    }
  });

  for (const launch of activeLaunches) {
    if (!launch.pairAddress) continue;

    try {
      const { data } = await axios.get(
        `https://api.dexscreener.com/latest/dex/pairs/bsc/${launch.pairAddress}`,
        { timeout: 5000 }
      );

      const pair = data.pair;
      if (!pair) continue;

      const snapshot = {
        launchId: launch.id,
        volume24h: pair.volume?.h24 ?? 0,
        priceUsd: parseFloat(pair.priceUsd ?? '0'),
        priceChange24h: pair.priceChange?.h24 ?? 0,
        liquidity: pair.liquidity?.usd ?? 0,
        holders: pair.holders ?? 0,
        recordedAt: new Date()
      };

      await db.launchSnapshot.create({ data: snapshot });

      const hoursLive = (Date.now() - new Date(launch.launchedAt).getTime()) / (1000 * 60 * 60);

      console.log(`[Tracker] ${launch.tokenName} | ${hoursLive.toFixed(0)}h live | vol $${snapshot.volume24h.toFixed(0)} | price $${snapshot.priceUsd.toFixed(6)} | liq $${snapshot.liquidity.toFixed(0)}`);

    } catch (err: any) {
      console.error(`[Tracker] Failed for ${launch.tokenName}:`, err.message);
    }
  }
}

export async function generateWeeklyReport(): Promise<string> {
  const oneWeekAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);

  const launches = await db.tokenLaunch.findMany({
    where: { launchedAt: { gte: oneWeekAgo }, status: 'live' },
    include: { snapshots: { orderBy: { recordedAt: 'desc' }, take: 1 } }
  });

  const successful = launches.filter(l => (l.snapshots[0]?.volume24h ?? 0) > 10000);
  const totalVolume = launches.reduce((sum, l) => sum + (l.snapshots[0]?.volume24h ?? 0), 0);
  const totalLiquidityDeployed = launches.reduce((sum, l) => sum + l.initialLiquidityUsdt, 0);

  const report = `
WEEKLY LAUNCH REPORT

Launches this week: ${launches.length}
Successful (over $10k volume): ${successful.length}
Success rate: ${((successful.length / launches.length) * 100).toFixed(1)}%
Total volume generated: $${totalVolume.toFixed(0)}
Total liquidity deployed: $${totalLiquidityDeployed.toFixed(0)}

Top performers:
${successful.slice(0, 3).map(l => `  ${l.tokenName} ($${l.ticker}) — $${(l.snapshots[0]?.volume24h ?? 0).toFixed(0)} volume`).join('\n')}

Signal type performance:
${generateSourceTypeBreakdown(launches)}
  `.trim();

  console.log(`[WeeklyReport]\n${report}`);
  return report;
}

function generateSourceTypeBreakdown(launches: any[]): string {
  const byType: Record<string, { count: number; avgVolume: number }> = {};

  for (const launch of launches) {
    if (!byType[launch.signalSource]) {
      byType[launch.signalSource] = { count: 0, avgVolume: 0 };
    }
    byType[launch.signalSource].count++;
    byType[launch.signalSource].avgVolume += (launch.snapshots[0]?.volume24h ?? 0);
  }

  return Object.entries(byType)
    .map(([type, data]) => `  ${type}: ${data.count} launches, avg $${(data.avgVolume / data.count).toFixed(0)} volume`)
    .join('\n');
}

export async function gatherXPhrases(): Promise<NarrativeCandidate[]> {
  if (!process.env.TWITTER_BEARER_TOKEN) return [];

  try {
    const headers = { Authorization: `Bearer ${process.env.TWITTER_BEARER_TOKEN}` };
    const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000).toISOString();

    const { data } = await axios.get(
      `https://api.twitter.com/2/tweets/search/recent`,
      {
        headers,
        params: {
          query: '(crypto OR BSC OR BNB OR memecoin) -is:retweet lang:en',
          max_results: 100,
          start_time: oneHourAgo,
          'tweet.fields': 'public_metrics,created_at'
        }
      }
    );

    const tweets = data.data ?? [];
    const phrases = extractEmergingPhrases(tweets);

    return phrases.map(phrase => ({
      keyword: phrase,
      description: `Emerging phrase on X with accelerating mention velocity`,
      sourceType: 'x_phrase' as const,
      rawData: {
        xPostsLastHour: tweets.filter((t: any) =>
          t.text.toLowerCase().includes(phrase.toLowerCase())
        ).length,
        sourceUrl: `https://twitter.com/search?q=${encodeURIComponent(phrase)}`,
        sourceCount: 1,
        btcChange4h: 0
      },
      detectedAt: new Date().toISOString()
    }));
  } catch {
    return [];
  }
}

function extractEmergingPhrases(tweets: any[]): string[] {
  const wordFreq: Record<string, number> = {};

  for (const tweet of tweets) {
    const words = tweet.text
      .split(/\s+/)
      .filter((w: string) => w.length > 4 && !w.startsWith('http') && !w.startsWith('@'))
      .map((w: string) => w.replace(/[^a-zA-Z0-9]/g, '').toLowerCase());

    for (const word of words) {
      wordFreq[word] = (wordFreq[word] ?? 0) + 1;
    }
  }

  return Object.entries(wordFreq)
    .filter(([word, count]) => count >= 5 && word.length > 4)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
    .map(([word]) => word);
}

export async function gatherAgentMilestones(): Promise<NarrativeCandidate[]> {
  const candidates: NarrativeCandidate[] = [];

  // Check for agents hitting trade milestones
  const agents = await db.agent.findMany({
    where: { isActive: true },
    include: {
      trades: { orderBy: { createdAt: 'desc' }, take: 100 }
    }
  });

  for (const agent of agents) {
    const tradeCount = agent.trades.length;
    const milestones = [10, 50, 100, 250, 500, 1000];

    if (milestones.includes(tradeCount)) {
      const winRate = agent.trades.filter((t: any) => t.pnl > 0).length / tradeCount;
      const totalPnl = agent.trades.reduce((sum: number, t: any) => sum + (t.pnl ?? 0), 0);

      if (winRate > 0.55 || totalPnl > 50) {
        candidates.push({
          keyword: agent.name,
          description: `Agent ${agent.name} hit ${tradeCount} trades with ${(winRate * 100).toFixed(0)}% win rate and $${totalPnl.toFixed(2)} PnL`,
          sourceType: 'agent_milestone',
          rawData: {
            agentId: agent.id,
            tradeCount,
            winRate,
            totalPnl,
            sourceCount: 2,
            btcChange4h: 0
          },
          detectedAt: new Date().toISOString()
        });
      }
    }
  }

  return candidates;
}
