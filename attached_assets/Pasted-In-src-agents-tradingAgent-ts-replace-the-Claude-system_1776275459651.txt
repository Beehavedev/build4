export function calculateEMA(prices: number[], period: number): number {
  const k = 2 / (period + 1)
  let ema = prices[0]
  for (let i = 1; i < prices.length; i++) {
    ema = prices[i] * k + ema * (1 - k)
  }
  return ema
}

export function calculateRSI(prices: number[], period: number): number {
  let gains = 0, losses = 0
  for (let i = 1; i <= period; i++) {
    const change = prices[prices.length - i] - prices[prices.length - i - 1]
    if (change > 0) gains += change
    else losses += Math.abs(change)
  }
  const avgGain = gains / period
  const avgLoss = losses / period
  if (avgLoss === 0) return 100
  const rs = avgGain / avgLoss
  return 100 - (100 / (1 + rs))
}

export function calculateMACD(prices: number[], fast: number, slow: number, signal: number) {
  const emaFast = calculateEMA(prices, fast)
  const emaSlow = calculateEMA(prices, slow)
  const macdLine = emaFast - emaSlow
  
  // Get recent MACD values to detect cross
  const recentMacd = prices.slice(-6).map((_, i, arr) => {
    const slice = prices.slice(0, prices.length - arr.length + i + 1)
    return calculateEMA(slice, fast) - calculateEMA(slice, slow)
  })
  
  const signalLine = calculateEMA(recentMacd, signal)
  const histogram = macdLine - signalLine
  
  const prevHistogram = recentMacd[recentMacd.length - 2] - signalLine
  let recentCross = null
  if (prevHistogram <= 0 && histogram > 0) recentCross = 'BULLISH CROSS'
  if (prevHistogram >= 0 && histogram < 0) recentCross = 'BEARISH CROSS'
  
  return { macdLine, signalLine, histogram, recentCross }
}

export function calculateBollingerBands(prices: number[], period: number, stdDev: number) {
  const slice = prices.slice(-period)
  const mid = slice.reduce((a, b) => a + b, 0) / period
  const variance = slice.reduce((a, b) => a + Math.pow(b - mid, 2), 0) / period
  const std = Math.sqrt(variance)
  return { upper: mid + stdDev * std, mid, lower: mid - stdDev * std }
}

export function calculateATR(ohlcv: { high: number[], low: number[], close: number[] }, period: number): number {
  const trs = []
  for (let i = 1; i < ohlcv.high.length; i++) {
    const tr = Math.max(
      ohlcv.high[i] - ohlcv.low[i],
      Math.abs(ohlcv.high[i] - ohlcv.close[i - 1]),
      Math.abs(ohlcv.low[i] - ohlcv.close[i - 1])
    )
    trs.push(tr)
  }
  return trs.slice(-period).reduce((a, b) => a + b, 0) / period
}

export function calculateADX(ohlcv: { high: number[], low: number[], close: number[] }, period: number): number {
  const atrs = [], plusDMs = [], minusDMs = []
  for (let i = 1; i < ohlcv.high.length; i++) {
    const upMove = ohlcv.high[i] - ohlcv.high[i - 1]
    const downMove = ohlcv.low[i - 1] - ohlcv.low[i]
    plusDMs.push(upMove > downMove && upMove > 0 ? upMove : 0)
    minusDMs.push(downMove > upMove && downMove > 0 ? downMove : 0)
    atrs.push(Math.max(
      ohlcv.high[i] - ohlcv.low[i],
      Math.abs(ohlcv.high[i] - ohlcv.close[i - 1]),
      Math.abs(ohlcv.low[i] - ohlcv.close[i - 1])
    ))
  }
  const slice = (arr: number[]) => arr.slice(-period).reduce((a, b) => a + b, 0) / period
  const atr = slice(atrs), plusDM = slice(plusDMs), minusDM = slice(minusDMs)
  const plusDI = (plusDM / atr) * 100
  const minusDI = (minusDM / atr) * 100
  const dx = Math.abs(plusDI - minusDI) / (plusDI + minusDI) * 100
  return dx
}

export function findRecentSwingHigh(ohlcv: { high: number[] }, lookback: number): number {
  return Math.max(...ohlcv.high.slice(-lookback))
}

export function findRecentSwingLow(ohlcv: { low: number[] }, lookback: number): number {
  return Math.min(...ohlcv.low.slice(-lookback))
}

export function calculateTrendAlignment(
  ema9: number, ema21: number, ema50: number, ema200: number, price: number
): number {
  let score = 0
  if (ema9 > ema21) score++
  if (price > ema50) score++
  if (price > ema200) score++
  if (ema21 > ema50) score++
  if (ema50 > ema200) score++
  return score
}

export function formatRecentCandles(
  ohlcv: { open: number[], high: number[], low: number[], close: number[], volume: number[] },
  count: number
): string {
  const len = ohlcv.close.length
  const lines = []
  for (let i = len - count; i < len; i++) {
    const bullish = ohlcv.close[i] > ohlcv.open[i]
    const change = ((ohlcv.close[i] - ohlcv.open[i]) / ohlcv.open[i] * 100).toFixed(2)
    lines.push(`  ${bullish ? '🟢' : '🔴'} O:${ohlcv.open[i].toFixed(2)} H:${ohlcv.high[i].toFixed(2)} L:${ohlcv.low[i].toFixed(2)} C:${ohlcv.close[i].toFixed(2)} (${change}%) Vol:${(ohlcv.volume[i] / 1000).toFixed(0)}k`)
  }
  return lines.join('\n')
}

export function buildAlignmentBar(alignment: {

export async function buildMemoryContext(agentId: string): Promise<string> {
  const memories = await db.agentMemory.findMany({
    where: { agentId },
    orderBy: { createdAt: 'desc' },
    take: 30
  })

  const corrections = memories.filter(m => m.type === 'correction')
  const observations = memories.filter(m => m.type === 'observation').slice(0, 10)
  const marketNotes = memories.filter(m => m.type === 'market_note').slice(0, 5)

  let context = ''

  if (corrections.length > 0) {
    context += `MISTAKES TO LEARN FROM (apply these lessons):\n`
    corrections.slice(0, 5).forEach(c => {
      context += `• ${c.content}\n`
    })
    context += '\n'
  }

  if (marketNotes.length > 0) {
    context += `MARKET PATTERNS NOTICED:\n`
    marketNotes.forEach(n => {
      context += `• ${n.content}\n`
    })
    context += '\n'
  }

  if (observations.length > 0) {
    context += `RECENT OBSERVATIONS:\n`
    observations.slice(0, 5).forEach(o => {
      context += `• ${o.content}\n`
    })
  }

  return context || 'No memories yet. Building knowledge from first trades.'
}

async function proactiveAlerts(agent: Agent, bot: Bot) {

  // Alert 1: Position at risk
  const positions = await getOpenPositions(agent.id)
  for (const pos of positions) {
    const distanceToLiq = Math.abs(pos.currentPrice - pos.liquidationPrice) / pos.currentPrice
    if (distanceToLiq < 0.05) {
      await bot.api.sendMessage(agent.userId,
        `⚠️ *Liquidation Warning* — ${agent.name}\n\n${pos.pair} ${pos.side} is within 5% of liquidation price.\n\nLiquidation: $${pos.liquidationPrice}\nCurrent: $${pos.currentPrice}\n\n[Close Position](close_${pos.id}) or add margin to protect.`,
        { parse_mode: 'Markdown' })
    }
  }

  // Alert 2: Strong signal detected even if no trade taken
  // (Sent when setupScore >= 8 but agent chose HOLD due to existing position)
  // This is set as a flag on the decision object and checked here

  // Alert 3: Daily summary (runs once per day at 00:00 UTC)
  const now = new Date()
  if (now.getUTCHours() === 0 && now.getUTCMinutes() < 2) {
    const todayTrades = await getTodayTrades(agent.id)
    const todayPnl = todayTrades.reduce((sum, t) => sum + (t.pnl || 0), 0)
    const wins = todayTrades.filter(t => t.pnl > 0).length
    
    if (todayTrades.length > 0) {
      await bot.api.sendMessage(agent.userId,
        `📊 *Daily Summary — ${agent.name}*\n\nTrades: ${todayTrades.length} (${wins}W/${todayTrades.length - wins}L)\nPnL: ${todayPnl >= 0 ? '+' : ''}$${todayPnl.toFixed(2)} USDT\nWin Rate Today: ${(wins / todayTrades.length * 100).toFixed(0)}%\n\nAll-time: ${agent.totalTrades} trades | ${agent.winRate.toFixed(0)}% win rate | +$${agent.totalPnl.toFixed(2)}`,
        { parse_mode: 'Markdown' })
    }
  }
}
