STEP 1 — Fetch klines for all 10 pairs in parallel
  const klineResults = await Promise.all(
    WATCHLIST.map(async symbol => {
      try {
        const [m15, h1, h4] = await Promise.all([
          getKlines(symbol, '15m', 100),
          getKlines(symbol, '1h', 100),
          getKlines(symbol, '4h', 100)
        ])
        return { symbol, m15, h1, h4 }
      } catch {
        return null
      }
    })
  )

STEP 2 — Score each pair DETERMINISTICALLY (no LLM)
  Run cheap TA calculations on each pair:
  
  function scoreSetup(klines15m, klines1h, klines4h): number {
    let score = 0
    
    const adx = calculateADX(klines1h)
    const rsi = calculateRSI(klines15m)
    const macd = calculateMACD(klines1h)
    const bb = calculateBB(klines15m)
    
    // Trending market
    if (adx > 25) score += 3
    else if (adx > 20) score += 1
    else return 0  // Ranging — skip entirely, score 0
    
    // RSI not overbought/oversold
    if (rsi > 40 && rsi < 60) score += 2
    else if (rsi > 30 && rsi < 70) score += 1
    
    // MACD momentum
    if (macd.histogram > 0 && macd.histogram > macd.prevHistogram) score += 2
    else if (macd.histogram < 0 && macd.histogram < macd.prevHistogram) score += 2
    
    // BB not squeezed
    const bbWidth = (bb.upper - bb.lower) / bb.middle
    if (bbWidth > 0.02) score += 1
    
    return score
  }

STEP 3 — Pick the single best pair
  const scored = klineResults
    .filter(Boolean)
    .map(r => ({
      ...r,
      score: scoreSetup(r.m15, r.h1, r.h4)
    }))
    .sort((a, b) => b.score - a.score)
  
  const best = scored[0]
  
  // Only proceed if score > 5 — otherwise HOLD everything
  if (!best || best.score < 5) {
    console.log('[Agent] No quality setup found across all pairs — HOLD')
    return
  }
  
  console.log(`[Agent] Best setup: ${best.symbol} score=${best.score}`)
