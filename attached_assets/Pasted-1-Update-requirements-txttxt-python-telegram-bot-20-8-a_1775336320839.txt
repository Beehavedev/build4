import os
from aster_dex.futures import Frenchie as AsterClient  # official SDK (Frenchie alias)

class AsterTrader:
    def __init__(self):
        self.client = AsterClient(
            key=os.getenv("ASTER_API_KEY"),
            secret=os.getenv("ASTER_API_SECRET"),
            recvWindow=10000,      # helps with timestamp issues
        )

    def get_balance(self):
        """Returns wallet + futures balance (USDT, etc.)"""
        return self.client.balance()

    def get_positions(self):
        """Current open positions with unrealized PNL"""
        return self.client.positionRisk()  # or self.client.position() if available

    def get_account_info(self):
        """Full account snapshot (balances + unrealized PNL)"""
        return self.client.accountWithJoinMargin()  # best for PNL overview

    def get_recent_trades(self, symbol="BTCUSDT", limit=10):
        """Last filled trades"""
        return self.client.userTrades(symbol=symbol, limit=limit)

    def get_income_history(self, symbol=None, limit=10):
        """Realized PNL, funding fees, commissions (full PNL history)"""
        params = {"limit": limit}
        if symbol:
            params["symbol"] = symbol
        return self.client.income(**params)

    def get_market_data(self, symbol="BTCUSDT"):
        """Latest ticker for quick price"""
        # SDK supports public endpoints too
        return self.client.ticker(symbol=symbol)  # or use klines below

    def get_klines(self, symbol="BTCUSDT", interval="1m", limit=100):
        """OHLCV data for TA/strategy"""
        return self.client.klines(symbol=symbol, interval=interval, limit=limit)

    def place_order(self, symbol, side, quantity, leverage=10, order_type="MARKET", price=None):
        """Place market or limit order"""
        params = {
            "symbol": symbol,
            "side": side.upper(),          # BUY or SELL
            "type": order_type.upper(),
            "quantity": quantity,
            "leverage": leverage,
        }
        if order_type.upper() == "LIMIT" and price:
            params["price"] = price
            params["timeInForce"] = "GTC"
        return self.client.new_order(**params)
