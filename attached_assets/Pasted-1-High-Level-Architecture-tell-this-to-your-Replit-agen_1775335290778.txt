aster_client.pypython
import os
from aster_dex.futures import Frenchie as AsterClient  # official SDK

class AsterTrader:
    def __init__(self):
        self.client = AsterClient(
            key=os.getenv("ASTER_API_KEY"),
            secret=os.getenv("ASTER_API_SECRET")
        )

    def get_balance(self):
        return self.client.balance()

    def get_positions(self):
        return self.client.position()

    def place_order(self, symbol, side, qty, leverage=10, order_type="MARKET"):
        # Example for perpetual futures
        return self.client.order(
            symbol=symbol,
            side=side,          # "BUY" or "SELL"
            type=order_type,
            quantity=qty,
            leverage=leverage
        )

    def get_market_data(self, symbol="BTCUSDT", interval="1m"):
        # Use public endpoints or SDK ticker/klines
        return self.client.ticker(symbol=symbol)

strategy.py (simple but powerful starter — upgrade later)python
import pandas as pd
import ta  # technical analysis library

def simple_ai_strategy(df: pd.DataFrame):
    # EMA + RSI example (you can replace with LLM call)
    df['ema8'] = ta.trend.EMAIndicator(df['close'], window=8).ema_indicator()
    df['ema21'] = ta.trend.EMAIndicator(df['close'], window=21).ema_indicator()
    df['rsi'] = ta.momentum.RSIIndicator(df['close']).rsi()

    if df['ema8'].iloc[-1] > df['ema21'].iloc[-1] and df['rsi'].iloc[-1] < 70:
        return "BUY"
    elif df['ema8'].iloc[-1] < df['ema21'].iloc[-1] and df['rsi'].iloc[-1] > 30:
        return "SELL"
    return "HOLD"

bot.py (Telegram interface + starts the agent)python
from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes
import asyncio
from agent import AutonomousAgent
import os

agent = AutonomousAgent()

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    agent.chat_id = update.effective_chat.id
    await update.message.reply_text("✅ Aster AI Agent is ready!\nUse /trade to start autonomous mode.")

async def trade(update: Update, context: ContextTypes.DEFAULT_TYPE):
    asyncio.create_task(agent.run())
    await update.message.reply_text("🤖 Autonomous trading STARTED on Aster DEX!")

async def stop(update: Update, context: ContextTypes.DEFAULT_TYPE):
    agent.running = False
    await update.message.reply_text("⛔ Trading stopped.")

async def status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    balance = agent.trader.get_balance()
    positions = agent.trader.get_positions()
    await update.message.reply_text(f"💰 Balance: {balance}\n📍 Positions: {positions}")

def main():
    app = Application.builder().token(os.getenv("TELEGRAM_TOKEN")).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("trade", trade))
    app.add_handler(CommandHandler("stop", stop))
    app.add_handler(CommandHandler("status", status))
    print("Telegram bot running...")
    app.run_polling()

if __name__ == "__main__":
    main()
