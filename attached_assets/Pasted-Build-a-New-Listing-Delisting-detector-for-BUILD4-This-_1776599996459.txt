import axios from 'axios'

// Track known pairs — detect when new ones appear
let knownPairs = new Set<string>()
let knownPairsLoadedAt = 0

interface ListingEvent {
  type: 'NEW_LISTING' | 'DELISTING' | 'REDUCE_ONLY'
  symbol: string
  detectedAt: number
  onboardDate?: number
  deliveryDate?: number
}

// Check Aster exchangeInfo every 60s for new/removed pairs
export async function checkForListingChanges(): Promise<ListingEvent[]> {
  try {
    const res = await axios.get(
      'https://fapi.asterdex.com/fapi/v1/exchangeInfo',
      { timeout: 8000 }
    )
    
    const symbols = res.data.symbols as any[]
    const events: ListingEvent[] = []
    const currentPairs = new Set<string>()
    
    for (const s of symbols) {
      if (s.quoteAsset !== 'USDT') continue
      currentPairs.add(s.symbol)
      
      // NEW LISTING detected
      if (knownPairs.size > 0 && !knownPairs.has(s.symbol)) {
        events.push({
          type: 'NEW_LISTING',
          symbol: s.symbol,
          detectedAt: Date.now(),
          onboardDate: s.onboardDate
        })
        console.log(`[Listing] 🚀 NEW LISTING DETECTED: ${s.symbol}`)
      }
      
      // REDUCE ONLY = delisting soon
      if (s.status === 'SETTLING' || s.status === 'BREAK') {
        if (knownPairs.has(s.symbol)) {
          events.push({
            type: 'REDUCE_ONLY',
            symbol: s.symbol,
            detectedAt: Date.now(),
            deliveryDate: s.deliveryDate
          })
          console.log(`[Listing] ⚠️ REDUCE ONLY (delisting): ${s.symbol}`)
        }
      }
    }
    
    // DELISTED — pair removed entirely
    for (const known of knownPairs) {
      if (!currentPairs.has(known)) {
        events.push({
          type: 'DELISTING',
          symbol: known,
          detectedAt: Date.now()
        })
        console.log(`[Listing] 🔴 DELISTED: ${known}`)
      }
    }
    
    knownPairs = currentPairs
    knownPairsLoadedAt = Date.now()
    return events
    
  } catch (err) {
    console.error('[Listing] exchangeInfo fetch failed:', err)
    return []
  }
}

// Get newly listed pairs from last 48 hours
// These have explosive volume — prime for agents to trade
export async function getRecentNewListings(): Promise<string[]> {
  try {
    const res = await axios.get(
      'https://fapi.asterdex.com/fapi/v1/exchangeInfo',
      { timeout: 8000 }
    )
    const symbols = res.data.symbols as any[]
    const cutoff = Date.now() - 48 * 60 * 60 * 1000  // 48h ago
    
    return symbols
      .filter(s => 
        s.quoteAsset === 'USDT' &&
        s.status === 'TRADING' &&
        s.onboardDate > cutoff
      )
      .map(s => s.symbol)
  } catch {
    return []
  }
}

import { checkForListingChanges, getRecentNewListings } 
  from './services/listingDetector'

// Run every 60 seconds — faster than any human
async function listingMonitor() {
  // Initialize known pairs on first run
  await checkForListingChanges()
  
  setInterval(async () => {
    const events = await checkForListingChanges()
    
    for (const event of events) {
      if (event.type === 'NEW_LISTING') {
        await handleNewListing(event)
      } else if (event.type === 'REDUCE_ONLY' || 
                 event.type === 'DELISTING') {
        await handleDelisting(event)
      }
    }
  }, 60_000)
}

async function handleNewListing(event: ListingEvent) {
  console.log(`[Listing] Processing new listing: ${event.symbol}`)
  
  // 1. Add to watchlist immediately
  // 2. Alert all active users via Telegram
  const activeUsers = await db.user.findMany({
    where: { agents: { some: { isActive: true } } },
    select: { telegramId: true }
  })
  
  for (const user of activeUsers) {
    try {
      await bot.api.sendMessage(
        user.telegramId,
        `🚀 NEW LISTING DETECTED\n\n` +
        `${event.symbol} just listed on Aster!\n\n` +
        `⚡ Your AI agents are now scanning this pair.\n` +
        `New listings often move 50-200% in the first hour.\n\n` +
        `📊 BUILD4 detected this before most humans.`
      )
    } catch {}
  }
  
  // 3. Add to dynamic watchlist so agents pick it up
  dynamicWatchlist.add(event.symbol)
  
  // 4. Log the listing event  
  console.log(`[Listing] Alerted ${activeUsers.length} users about ${event.symbol}`)
}

async function handleDelisting(event: ListingEvent) {
  console.log(`[Listing] Processing delisting: ${event.symbol}`)
  
  // 1. Find any open positions in this pair
  const openPositions = await db.trade.findMany({
    where: { 
      pair: event.symbol,
      status: 'OPEN'
    },
    include: { agent: { include: { user: true } } }
  })
  
  // 2. Alert users with open positions immediately
  for (const pos of openPositions) {
    try {
      await bot.api.sendMessage(
        pos.agent.user.telegramId,
        `⚠️ DELISTING ALERT — ${event.symbol}\n\n` +
        `Aster is delisting this pair.\n` +
        `Your agent has an open position.\n\n` +
        `Action: Your agent will close this position\n` +
        `automatically before delisting.\n\n` +
        `Stay calm — funds are safe.`
      )
    } catch {}
  }
  
  // 3. Remove from watchlist
  dynamicWatchlist.delete(event.symbol)
}

// Dynamic watchlist — starts with base pairs, 
// adds new listings, removes delistings
export const dynamicWatchlist = new Set<string>([
  'BTCUSDT', 'ETHUSDT', 'SOLUSDT', 'BNBUSDT',
  'XRPUSDT', 'DOGEUSDT', 'ASTERUSDT', 'AVAXUSDT',
  'LINKUSDT', 'ARBUSDT'
])

listingMonitor()  // Start on server launch
